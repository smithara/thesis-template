\makeglossaries

% A simpler macro if you only want a simple glossary
%  with no linking, or defining terms in-text
\newcommand{\glossterm}[2]{
    \newglossaryentry{#1}{
        name=#1,
        description={#2}
    }
}

\glossterm{MCT}{My complicated term}
\glossterm{VHF}{Very hard formula}

% (normally they only appear when linked to in the main text)
